│ 5.2 補間アルゴリズム例                                                                                               │
│                                                                                                                      │
│ // 累月値（11月: 2,465.7億円）から12か月を生成                                                                       │
│ function generateMonthlyFromCumulative(cumulative, months = 11) {                                                    │
│   const seasonal = [0.08, 0.07, 0.09, 0.10, 0.09, 0.08,                                                              │
│                     0.10, 0.09, 0.08, 0.07, 0.08, 0.07]; // 季節調整                                                 │
│   const baseValue = cumulative / seasonal.slice(0, months).reduce((a,b) => a+b);                                     │
│   return seasonal.map(s => baseValue * s);                                                                           │
│ }                                                                                                                    │
│                                                                                                                      │
│ // 計画値生成（期央予算210.0億円を按分）                                                                             │
│ function generatePlan(actualMonthly, planTotal) {                                                                    │
│   const actualSum = actualMonthly.reduce((a,b) => a+b);                                                              │
│   return actualMonthly.map(v => (v / actualSum) * planTotal);                                                        │
│ }                                                                                                                    │
│                                                                                                                      │
│ // 前年値生成（前年差+19.3億円を差し引き）                                                                           │
│ function generatePriorYear(actualMonthly, yoyDelta) {                                                                │
│   const deltaPerMonth = yoyDelta / actualMonthly.length;                                                             │
│   return actualMonthly.map(v => v - deltaPerMonth);                                                                  │
│ }                                                                                                                    │

     5.2 補間アルゴリズム例

     // 累月値（11月: 2,465.7億円）から12か月を生成
     function generateMonthlyFromCumulative(cumulative, months = 11) {
       const seasonal = [0.08, 0.07, 0.09, 0.10, 0.09, 0.08,
                         0.10, 0.09, 0.08, 0.07, 0.08, 0.07]; // 季節調整
       const baseValue = cumulative / seasonal.slice(0, months).reduce((a,b) => a+b);
       return seasonal.map(s => baseValue * s);
     }

     // 計画値生成（期央予算210.0億円を按分）
     function generatePlan(actualMonthly, planTotal) {
       const actualSum = actualMonthly.reduce((a,b) => a+b);
       return actualMonthly.map(v => (v / actualSum) * planTotal);
     }

     // 前年値生成（前年差+19.3億円を差し引き）
     function generatePriorYear(actualMonthly, yoyDelta) {
       const deltaPerMonth = yoyDelta / actualMonthly.length;
       return actualMonthly.map(v => v - deltaPerMonth);
     }
